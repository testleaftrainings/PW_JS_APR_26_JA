1.In JavaScript, strings are immutable, which means once a string is created, its actual value in memory cannot be changed.

So when we think we are modifying a string, JavaScript is not changing the existing string. Instead, it creates a completely new string and updates the variable reference to point to that new value.

let str = "hello";
str = "world";

Here, JavaScript does not modify "hello" into "world".
It creates a new string "world" and str now points to that new string.

let logs = "ERROR: 404 [Time: 12:00] | INFO: Login Success [Time: 12:01] | ERROR: 500 [Time: 12:05]";

let parts = logs.split("|");

let result = [];

for(let log of parts){

    if(log.includes("ERROR")){

        let start = log.indexOf("[Time:") + 7;
        let end = log.indexOf("]");

        let time = log.slice(start, end).trim();

        result.push(time);
    }
}

console.log(result);
